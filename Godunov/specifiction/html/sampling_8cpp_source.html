16double f_l(double p, double* statesl)
17{
19 double ans;
20 double pl, rhol, Al, Bl;
21 pl = statesl[2];
22 rhol = statesl[0];
23 Al = 2 / ( (gamma+1) * rhol);
24 Bl = (gamma - 1) * pl / (gamma + 1);
25 if (p > pl)
26 {
27 ans = Al / (p + Bl);
28 ans = sqrt(ans);
29 ans = ans * (p - pl);
30 }
31 else{
32 double temp, al;
33 temp = (gamma - 1)/(2 * gamma);
34 rhol = statesl[0];
35 al = sqrt( gamma * pl / rhol);
36
37 ans = p / pl;
38 ans = pow(ans, temp);
39 ans = 2 * al * (ans - 1)/(gamma - 1);
40 }
41 return ans;
42}

52 Br = (gamma - 1) * pr / (gamma + 1);
53 if ( p > pr)
54 {
55 ans = Ar / ( p + Br );
56 ans = sqrt(ans);
57 ans = ans * (p - pr);

60 double temp, ar;

62 ar = sqrt( gamma * pr / rhor);

78double f_ldp(double p, double* statesl)
79{
80 double ans;
81 double pl, rhol, Al, Bl;
82 pl = statesl[2];
83 rhol = statesl[0];
84 Al = 2 / ( (gamma+1) * rhol);
85 Bl = (gamma - 1) * pl / (gamma + 1);
86 if (p > pl)
87 {
88 ans = 1 - (p - pl)/(2 * (Bl + p));
89 ans = ans * sqrt(Al/(Bl + p));
90 }
91 else{
92 double temp, al;
93 temp = (gamma + 1)/(2 * gamma);
94 al = sqrt(gamma * pl / rhol);
95
96 ans = p/pl;
97 ans = pow(ans,-temp);
98 ans = ans / (rhol * al);
99 }
100 return ans;
101}

108 rhor = statesr[0];

110 Br = (gamma - 1) * pr / (gamma + 1);
111 if (p > pr)
112 {
113 ans = 1 - (p - pr)/(2 * (Br + p));
114 ans = ans * sqrt(Ar/(Br + p));
115 }

117 double temp, ar;
118 temp = (gamma + 1)/(2 * gamma);
119 ar = sqrt(gamma * pr / rhor);

122 ans = pow(ans,-temp);
123 ans = ans / (rhor * ar);
124 }

128double getpstar(double* statesl, double* statesr)
129{
130 double pstar, temp;
131 double fp, fdp, e, time;
132 double pl, pr;
133 pl = statesl[2];
134 pr = statesr[2];
135 e = 1e-6;
136 time = 1;
138 /*
139 double ul, ur, rhol, rhor, al, ar;
140 ul = statesl[1];
141 ur = statesr[1];
142 rhol = statesl[0];
143 rhor = statesr[0];
144 al = sqrt(gamma * pl / rhol);
145 ar = sqrt(gamma * pr / rhor);
146 pstar = al+ar - 0.5*(gamma-1)*(ur-ul);
147 pstar = pstar / (pow(al/pl,(gamma-1)/(2*gamma)) + pow(ar/pr,(gamma-1)/(2*gamma)));
148 pstar = pow(pstar,2*gamma/(gamma-1));
149 //*/
150
151 //** Initial pstar with mean value */
152 pstar = 0.5 * (pl + pr);
153 temp = 0;
154 while(abs(pstar - temp)>e)
155 {
156 time++;
157 if (time > 1e6)
158 {
159 break;
160 }
161
162 if (pstar <=0)
163 {
164 pstar = 1e-5;
165 }
166 temp = pstar;
167 fp = f(temp,statesl, statesr);
168 fdp = f_ldp(temp,statesl) + f_rdp(temp,statesr);
169 pstar = temp - fp / fdp;
170
171 }
172 return pstar;
173}

198double getrhostarl(double pstar, double* statesl, wavetype leftwave)
199{
200 double rhostarl;
201 double rhol, pl;
202 rhol = statesl[0];
203 pl = statesl[2];
204 if (leftwave == rarefaction)
205 {
206 rhostarl = pstar / pl;
207 rhostarl = rhol * pow(rhostarl,1.0/gamma);
208 }
209 else{
210 rhostarl = pstar / pl + (gamma - 1) / (gamma + 1);
211 rhostarl = rhostarl / ( ( (gamma-1)/(gamma+1) ) * ( pstar / pl) + 1 );
212 rhostarl = rhol * rhostarl;
213 }
214 return rhostarl;

236double* sampling(double x, double t, double* statesl, double* statesr)
237{
238 double* ans = new double[4];
239 wavetype lwave, rwave;
240 double pstar, ustar, rhostarl, rhostarr;
241 double rhot, ut, pt;
242 double rhol, rhor, ul, ur, pl, pr, al, ar;
243 double dll, dlr, dmid, drl, drr;
244
245 rhol = statesl[0];
246 rhor = statesr[0];
247 ul = statesl[1];
248 ur = statesr[1];
249 pl = statesl[2];
250 pr = statesr[2];
251 al = sqrt(gamma * pl / rhol);
252 ar = sqrt(gamma * pr / rhor);
253
254 if (rhol == rhor )
255 {
256 ans[0] = rhol;
257 ans[1] = ul;
258 ans[2] = pl;
259 return ans;
260 }
261 pstar = getpstar(statesl,statesr);
262 ustar = getustar(pstar,statesl, statesr);
263
264 lwave = identifyNonlinearWave(pl,statesl,statesr);
265 rwave = identifyNonlinearWave(pr,statesl,statesr);
266 rhostarl = getrhostarl(pstar,statesl,lwave);
267 rhostarr = getrhostarr(pstar,statesr,rwave);
268
269
270 if (lwave == shock)
271 {
272 double s;
273 s = (gamma + 1) * pstar / (2 * gamma * pl) + (gamma - 1)/(2 * gamma);
274 s = ul - al * sqrt(s);
275
276 dll = s;
277 dlr = s;
278 }
279 else{
280 double astarl, shl, stl;
281 astarl = al * pow(pstar/pl,(gamma-1)/(2*gamma));
282
283 shl = ul - al;
284 stl = ustar - astarl;
285
286 dll = shl;
287 dlr = stl;
288 }
289
290 dmid = ustar;
291
292 if (rwave == shock)
293 {
294 double s;
295 s = (gamma + 1) * pstar/(2 * gamma * pr) + (gamma - 1)/(2 * gamma);
296 s = ur + ar * sqrt(s);
297
298 drl = s;
299 drr = s;
300 }
301 else{
302 double astarr, shr, str;
303 astarr = ar * pow(pstar/pr, (gamma-1)/(2*gamma));
304 shr = ur + ar;
305 str = ustar + astarr;
306
307 drl = str;
308 drr = shr;
309 }
310
311
312 if (x/t <= dll)
313 {
314 rhot = rhol;
315 ut = ul;
316 pt = pl;
317 }
318 else if (x/t <= dlr && lwave == rarefaction)
319 {
320 rhot = 2/(gamma + 1) + (gamma-1)*(ul - x/t)/( (gamma+1) * al );
321 rhot = rhol * pow(rhot,2/(gamma-1));
322 ut = 2 * (al + (gamma-1) * ul /2 + x/t) /(gamma + 1);
323 pt = 2/(gamma + 1) + (gamma-1)*(ul - x/t) /( (gamma + 1) * al );
324 pt = pl * pow(pt,2*gamma/(gamma-1));
325 }
326 else if (x/t <= dmid)
327 {
328 rhot = rhostarl;
329 ut = ustar;
330 pt = pstar;
331 }
332 else if (x/t <= drl)
333 {
334 rhot = rhostarr;
335 ut = ustar;
336 pt = pstar;
337 }
338 else if (x/t <= drr && rwave == rarefaction)
339 {
340 rhot = 2/(gamma+1) - (gamma-1)*(ur - x/t)/( (gamma+1)*ar );
341 rhot = rhor * pow(rhot, 2/(gamma-1));
342 ut = 2 * (-ar + (gamma - 1) * ur / 2 + x/t) / (gamma + 1);
343 pt = 2/(gamma+1) - (gamma-1)*(ur - x/t)/( (gamma+1)*ar );
344 pt = pr * pow(pt, 2*gamma/(gamma-1));
345 }
346 else
347 {
348 rhot = rhor;
349 ut = ur;
350 pt = pr;
351 }
352
353 ans[0] = rhot;
354 ans[1] = ut;
355 ans[2] = pt;
356 return ans;
357}
