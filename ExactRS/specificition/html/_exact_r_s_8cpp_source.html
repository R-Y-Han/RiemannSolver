182double f_l(double p)
183{
184 double ans;
185 if (p > pl)
186 {
187 ans = Al / (p + Bl);
188 ans = sqrt(ans);
189 ans = ans * (p - pl);
190 }
191 else{
192 double temp;
193 temp = (gamma - 1)/(2 * gamma);
194
195 ans = p / pl;
196 ans = pow(ans, temp);
197 ans = 2 * al * (ans - 1)/(gamma - 1);
198 }
199 return ans;
200}

202double f_r(double p)
203{
204 double ans;
205 if ( p > pr)
206 {
207 ans = Ar / ( p + Br );
208 ans = sqrt(ans);
209 ans = ans * (p - pr);
210 }
211 else{
212 double temp;
213 temp = (gamma - 1) / (2 * gamma);
214
215 ans = p / pr;
216 ans = pow(ans, temp);
217 ans = 2 * ar * (ans - 1)/ (gamma - 1);
218 }
219 return ans;
220}

229double f_ldp(double p)
230{
231 double ans;
232
233 if (p > pl)
234 {
235 ans = 1 - (p - pl)/(2 * (Bl + p));
236 ans = ans * sqrt(Al/(Bl + p));
237 }
238 else{
239 double temp;
240 temp = (gamma + 1)/(2 * gamma);
241
242 ans = p/pl;
243 ans = pow(ans,-temp);
244 ans = ans / (rhol * al);
245 }
246 return ans;
247}

251 double ans;
252
253 if (p > pr)

255 ans = 1 - (p - pr)/(2 * (Br + p));

259 double temp;

261
262 ans = p/pr;
263 ans = pow(ans,-temp);
264 ans = ans / (rhor * ar);

269double getpstar()
270{
271 double pstar, temp;
272 double fp, fdp, e, time;
273 e = 1e-6;
274 time = 1;
275 //pstar = al+ar - 0.5*(gamma-1)*(ur-ul);
276 //pstar = pstar / (pow(al/pl,(gamma-1)/(2*gamma)) + pow(ar/pr,(gamma-1)/(2*gamma)));
277 //pstar = pow(pstar,2*gamma/(gamma-1));
278 pstar = 0.5 * (pl + pr);
279 temp = 0;
280 while(abs(pstar - temp)>e)
281 {
282 time++;
283 if (time > 1e6)
284 {
285 break;
286 }
287
288 temp = pstar;
289 fp = f(temp);
290 fdp = f_ldp(temp) + f_rdp(temp);
291 pstar = temp - fp / fdp;
292 }
293
294 return pstar;
295}

317double getrhostarl(double pstar, wavetype leftwave)
318{
319 double rhostarl;
320 if (leftwave == rarefaction)
321 {
322 rhostarl = pstar / pl;
323 rhostarl = rhol * pow(rhostarl,1.0/gamma);
324 }
325 else{
326 rhostarl = pstar / pl + (gamma - 1) / (gamma + 1);
327 rhostarl = rhostarl / ( ( (gamma-1)/(gamma+1) ) * ( pstar / pl) + 1 );
328 rhostarl = rhol * rhostarl;
329 }
330 return rhostarl;
331}
